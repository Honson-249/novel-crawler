#### JSON API 详情获取（兜底）

```python
# spider.py:670-730
async def _fetch_detail_via_api(self, detail_url, language):
    # 1. 从 URL 提取 UUID
    match = re.search(r'/shorts/([0-9a-f-]{36})', detail_url)
    short_id = match.group(1)

    # 2. 构造 Next.js API URL
    api_url = self.config.next_data_url(self._build_id, language, short_id)
    # 格式：/_next/data/{buildId}/{lang}/shorts/{uuid}.json

    # 3. HTTP 请求获取详情
    async with session.get(api_url) as resp:
        data = await resp.json()
        movie = data.get("pageProps", {}).get("movieDetails", {}).get("movie", {})
        return {"synopsis": movie.get("description", "")}
```

#### 剧集详情 API（含重定向跟随）

```python
# api_client.py:194-268
async def fetch_drama_detail(self, language, book_id):
    # 1. 检查 slug 缓存
    cached_slug = self._slug_cache.get(f"{language}:{book_id}")
    if cached_slug:
        # 直接用缓存 slug 请求
        url = f"{BASE_URL}/_next/data/{build_id}/{language}/movie/{cached_slug}.json"

    # 2. 用 book_id 请求，预期服务端返回 __N_REDIRECT
    url = f"{BASE_URL}/_next/data/{build_id}/{language}/movie/{book_id}.json"
    data = await self._get_json(url)

    # 3. 跟随重定向获取正确 slug
    if "__N_REDIRECT" in data:
        redirect_path = data["__N_REDIRECT"]
        real_slug = redirect_path.split("/movie/")[-1]
        self._slug_cache[f"{language}:{book_id}"] = real_slug

        # 用正确 slug 重新请求
        url2 = f"{BASE_URL}/_next/data/{build_id}/{language}/movie/{real_slug}.json"
        return await self._get_json(url2)
```

#### 分类算法

```python
# pipeline/reelshort_clean.py
def classify_tags(tag_list, actors_ref, actresses_ref, identity_ref, story_beat_ref):
    actors_tags = []
    actresses_tags = []
    identity_tags = []
    story_beat_tags = []
    genre_tags = []

    for tag in tag_list:
        if tag in actors_ref:
            actors_tags.append(tag)
        elif tag in actresses_ref:
            actresses_tags.append(tag)
        elif tag in identity_ref:
            identity_tags.append(tag)
        elif tag in story_beat_ref:
            story_beat_tags.append(tag)
        else:
            genre_tags.append(tag)

    return {
        "actors_tags": actors_tags,
        "actresses_tags": actresses_tags,
        "identity_tags": identity_tags,
        "story_beat_tags": story_beat_tags,
        "genre_tags": genre_tags,
    }
```

## 翻译服务

### DramaShorts 翻译服务

**翻译字段**：`board_name`, `series_title`, `synopsis`

**复用策略**：
1. URL 缓存：同语言下相同 `detail_url` 直接复用
2. board_name 预翻译：并发前统一翻译所有唯一榜单名

```python
# services/dramashort_translate_service.py
async def _translate_group(group, client, url_cache, board_cache):
    for record in group:
        if detail_url in url_cache:
            # 命中 URL 缓存，直接复用 series_title/synopsis
            cached = url_cache[detail_url]
            record.update(cached)

        # board_name 从预翻译缓存填充
        if board_name in board_cache:
            record["board_name"] = board_cache[board_name]

    # 构建 LLM payload（不含 board_name）
    payloads = [{"series_title": ..., "synopsis": ...}, ...]
    translated_list = await client.translate_records_batch(payloads)
```

### ReelShort 翻译服务

**翻译字段**：
- 纯文本：`series_title`, `synopsis`
- JSON 数组：`actors_tags`, `actresses_tags`, `identity_tags`, `story_beat_tags`, `genre_tags`

**优化策略**：
1. URL 缓存：同语言下相同 `detail_url` 直接复用
2. 标签缓存：同一次任务内相同标签只翻译一次
3. 标签对照表：从 `reelshort_tags` 预加载已翻译标签

```python
# services/reelshort_translate_service.py
async def _translate_group(group, client, ctx):
    # ctx.tag_cache 包含：
    #   - 从 reelshort_tags 预加载的标签译名
    #   - 本次任务中已翻译的标签

    for record in group:
        if detail_url in ctx.url_cache:
            # URL 缓存命中
            record.update(ctx.url_cache[detail_url])

        # 构建 payload 时过滤已缓存标签
        for field in JSON_TAG_FIELDS:
            tags = json.loads(record[field])
            new_tags = [t for t in tags if t not in ctx.tag_cache]
            if new_tags:
                payload[field] = json.dumps(new_tags)

    translated_list = await client.translate_records_batch(payloads)
```

## 两个短剧爬虫对比

| 特性 | DramaShorts | ReelShort |
|------|-------------|-----------|
| **爬取方式** | HTTP 获取 `__NEXT_DATA__` + Playwright 兜底 | 全程 `/_next/data/` API |
| **语言支持** | 单语言 (en) | 19 种语言 |
| **榜单结构** | 固定 7 个榜单（通过 block_id 映射） | 4 个 Tab（Actors/Actresses/Identities/Story Beats） |
| **标签分类** | 无 | 5 类（交叉比对参照集） |
| **翻译服务** | 支持（board_name, series_title, synopsis） | 支持（含 JSON 标签数组） |
| **断点续爬** | 无 | 支持（记录已爬页码） |
| **速度** | 中等（Playwright 兜底时较慢） | 快（10x 于 Playwright） |
| **详情页缓存** | 内存 + DB 三层缓存 | 内存 + DB + slug 缓存 |
| **数据清洗** | 基础字段清洗 | 标签分类 + 数值换算 |
